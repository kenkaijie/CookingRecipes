\begin{recipe}[ChilliConCarne]{Chilli Con Carne}{4 Portions}{1 hour}
    \freeform Simple chilli recipe with beans.
    \ingredient[\fr14]{tsp}{smoked paprika}
    \ingredient[1\fr12]{tsp}{paprika}
    \ingredient[\fr14]{tsp}{chilli powder}
    \ingredient[1]{tsp}{ground cumin seed}
    \ingredient[1]{tsp}{ground coriander}
    \ingredient[2]{tbsp}{plain flour}
    In a small bowl, mix dry ingredients to form a spice mix.
    \ingredient[2]{tbsp}{olive oil}
    \ingredient[1]{}{onion, finely chopped}
    \ingredient[2]{cloves}{garlic, finely chopped}
    In a small pot, heat oil and cook onions and garlic until slightly browned.
    \ingredient[500]{g}{beef mince}
    Add in beef mince and cook until seared. Add in the spice mix and cook briefly on low heat. Keep mix from sticking to bottom of pan.
    \ingredient[500]{ml}{chicken stock}
    Add in chicken stock to deglaze the pot.
    \ingredient[400]{g}{canned diced tomatoes}
    \ingredient[2]{tsp}{dried basil}
    \ingredient[2]{tsp}{dried oregano}
    \ingredient[2]{tbsp}{tomato paste}
    \ingredient[1]{}{bay leaf}
    \ingredient[400]{g}{red kidney beans}
    Add in diced tomatoes, tomato paste, beans, and herbs to the pot. Reduce heat to low and simmer for 40 minutes.
    \ingredient[2]{}{tomatoes, diced}
    \ingredient[1]{}{avocado}
    \ingredient[1]{}{bag of corn chips}
    \ingredient[]{}{sour cream}
    Once cooked, serve with helping of sour cream, corn chips and freshly diced whole tomatoes and avocado.
    \freeform\hrulefill
\end{recipe}
